<script>
const SUBJECTS=[
  {id:'calculo',label:'Cálculo Integral',color:'#e8c547',sym:'∫'},
  {id:'topologia',label:'Topología',color:'#7eb8f7',sym:'𝒯'},
  {id:'numeros',label:'Teoría de Números',color:'#f7a07e',sym:'ℕ'},
  {id:'algebra',label:'Álgebra Lineal',color:'#9de89a',sym:'⊗'},
];
const KEY='math-log-v1';
let sel=SUBJECTS[0].id;
let entries=[];
try{entries=JSON.parse(localStorage.getItem(KEY)||'[]');}catch(e){}

function save(){localStorage.setItem(KEY,JSON.stringify(entries));}

function fmt(h){
  const hh=Math.floor(h),mm=Math.round((h-hh)*60);
  return mm===0?`${hh}h`:`${hh}h ${mm}m`;
}
function today(){return new Date().toISOString().slice(0,10);}
function fmtDate(iso){const[y,m,d]=iso.split('-');return`${d}/${m}/${y}`;}

function totals(){
  const t={};
  SUBJECTS.forEach(s=>t[s.id]=entries.reduce((a,e)=>e.subject===s.id?a+e.hours:a,0));
  return t;
}

function renderCards(){
  const t=totals();
  const grand=Object.values(t).reduce((a,b)=>a+b,0);
  document.getElementById('grand').textContent=fmt(grand);
  document.getElementById('cards').innerHTML=SUBJECTS.map(s=>`
    <div class="card ${t[s.id]>0?'active':''}" style="--c:${s.color}">
      <span class="sym" style="color:${s.color}">${s.sym}</span>
      <div>
        <div class="card-label">${s.label}</div>
        <div class="card-hours">${fmt(t[s.id])}</div>
      </div>
    </div>`).join('');
}

function renderChips(){
  document.getElementById('chips').innerHTML=SUBJECTS.map(s=>`
    <button class="chip${sel===s.id?' sel':''}"
      style="color:${s.color};border-color:${s.color}55;${sel===s.id?`background:${s.color};color:#0f0f10;`:''}  "
      onclick="setSel('${s.id}',this)">
      ${s.sym} ${s.label}
    </button>`).join('');
}

function setSel(id){sel=id;renderChips();}

function addEntry(){
  const h=parseFloat(document.getElementById('inp-h').value);
  if(!h||h<=0||h>24)return;
  const n=document.getElementById('inp-n').value.trim();
  entries.unshift({id:Date.now(),date:today(),subject:sel,hours:h,note:n});
  save();
  document.getElementById('inp-h').value='';
  document.getElementById('inp-n').value='';
  renderCards();
  renderHistory();
  // flash card
  const cards=document.querySelectorAll('.card');
  const idx=SUBJECTS.findIndex(s=>s.id===sel);
  if(cards[idx]){cards[idx].classList.add('pop');setTimeout(()=>cards[idx].classList.remove('pop'),400);}
}

function delEntry(id){
  entries=entries.filter(e=>e.id!==id);
  save();renderCards();renderHistory();
}

function renderHistory(){
  const grouped={};
  entries.forEach(e=>{if(!grouped[e.date])grouped[e.date]=[];grouped[e.date].push(e);});
  const dates=Object.keys(grouped).sort((a,b)=>b.localeCompare(a));
  const el=document.getElementById('history-list');
  if(!dates.length){el.innerHTML='<div class="empty">Sin registros aún</div>';return;}
  el.innerHTML=dates.map(date=>{
    const dayTotal=grouped[date].reduce((s,e)=>s+e.hours,0);
    const rows=grouped[date].map(e=>{
      const s=SUBJECTS.find(x=>x.id===e.subject);
      return`<div class="entry" style="border-color:${s.color}22">
        <span class="entry-sym" style="color:${s.color}">${s.sym}</span>
        <div class="entry-info">${s.label}${e.note?`<span class="entry-note">${e.note}</span>`:''}
        </div>
        <span class="entry-h" style="color:${s.color}">${fmt(e.hours)}</span>
        <button class="del" onclick="delEntry(${e.id})">✕</button>
      </div>`;
    }).join('');
    return`<div class="date-group">
      <div class="date-head">${fmtDate(date)} — ${fmt(dayTotal)}</div>
      ${rows}
    </div>`;
  }).join('');
}

function showTab(id,btn){
  document.getElementById('tab-log').style.display=id==='log'?'':'none';
  document.getElementById('tab-history').style.display=id==='history'?'':'none';
  document.querySelectorAll('.tab').forEach(t=>t.classList.remove('on'));
  btn.classList.add('on');
}

// Enter key
document.addEventListener('keydown',e=>{if(e.key==='Enter')addEntry();});

renderCards();
renderChips();
renderHistory();
</script>
